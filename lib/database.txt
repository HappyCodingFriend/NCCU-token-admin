CREATE TABLE `order` (
  `address` varchar(55) NOT NULL,
  `owner` varchar(55) NOT NULL,
  `finish` tinyint(4) NOT NULL DEFAULT '0',
  PRIMARY KEY (`address`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `point` (
  `address` varchar(65) NOT NULL,
  `owner` varchar(65) NOT NULL,
  `valid` tinyint(4) NOT NULL DEFAULT '1',
  PRIMARY KEY (`address`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `transaction` (
  `txhash` varchar(80) NOT NULL,
  `from_addr` varchar(65) NOT NULL,
  `point` varchar(65) NOT NULL,
  PRIMARY KEY (`txhash`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
